#!/bin/bash
time=`date +%H:%M:%S`
msg=`scrot -s "$time.png" -e "curl -sF 'name=@$time.png' http://eleveni386.7axu.com/Image/"`
gxmessage -center -title "eleveni386.7axu.com" $msg
